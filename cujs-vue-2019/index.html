Putting it all together

Making the arcs reactive

Only requires two small changes!

const vm = new Vue({
  el: "#app",
  data: {
    percent: 25
  },
  computed: {
    arcOffset() {
      return 100 - this.percent;
    }
  }
});

Vue has computed property - derived values that let us keep templates clean and expressive

But it's pretty janky

Fixing the text formatting, instance

const formatter = new Intl.NumberFormat(
  navigator.language,
  {style: 'percent'}
);
const vm = new Vue({
  el: "#app",
  data: {
    percent: 25
  },
  computed: {
  	arcOffset() {
    	return 100 - this.percent;
    }
  },
  methods: {
    sanitizePercent(event) {
      const value = Number(event.target.value);
      if (isNaN(value)) {
      	this.percent = 0;
        return;
      }
      this.percent = Math.min(Math.max(value, 0), 100);
    }
  },
  filters: {
  	toPercentString(value) {
    	return formatter.format(value / 100);
    }
  }
});

We'll use the builtin NumberFormat method to render a locale-appropriate string

Adding some color, instance

const formatter = new Intl.NumberFormat(
  navigator.language,
  {style: 'percent'}
);
const colorMap = chroma.scale(
  ["green", "green", "yellow", "red"]
).domain(
  [0, 0.5, 0.66, 1]
);
const vm = new Vue({
  el: "#app",
  data: {
    percent: 25,
    arcOffset: 75
  },
  computed: {
    dataColor() {
      return colorMap(this.percent / 100);
    }
  },
  watch: {
    percent(newOffset) {
      TweenLite.to(this.$data, 1 / 4, {
        arcOffset: 100 - newOffset,
        ease: "linear"
      });
    }
  },
  methods: {
    sanitizePercent(event) {
      const value = Number(event.target.value);
      if (isNaN(value)) {
      	this.percent = 0;
        return;
      }
      this.percent = Math.min(Math.max(value, 0), 100);
    }
  },
  filters: {
  	toPercentString(value) {
    	return formatter.format(value / 100);
    }
  }
});
