    Gulp 

    var gulp = require('gulp');
var sass = require('gulp-sass');
var minifyCss = require('gulp-minify-css');
var rename = require('gulp-rename');

//declare the task
gulp.task('sass', function(done) {
  gulp.src('./scss/ionic.app.scss')
    .pipe(sass())
    .pipe(gulp.dest('./www/css/'))
    .pipe(minifyCss())
    .pipe(rename({ extname: '.min.css' }))
    .pipe(gulp.dest('./www/css/'))
    .on('end', done);
});

    Webpack

    Capable of transforming, bundling, or packaging just about any resource or asset.

    Lets you require('modules') in the browser by bundling up all of your dependencies.

    Rollup

    Statically analyses your code, and your dependencies, and includes the bare minimum in your bundle.
